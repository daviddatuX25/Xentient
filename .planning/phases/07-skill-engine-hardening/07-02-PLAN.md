# Plan 07-02: Generic MQTT Event Bridge

## Goal
Replace hardcoded MQTT event routing in `core.ts` with a configurable event bridge. Any MQTT topic can be mapped to a named event that skills can trigger on, eliminating the need to modify `core.ts` for new sensor types or event sources.

```typescript
import { EventEmitter } from 'events';
import pino from 'pino';
import { MqttClient } from './MqttClient';
import { SpaceManager } from '../engine/SpaceManager';
import { ModeManager } from '../engine/ModeManager';
import { PERIPHERAL_IDS } from '../shared/contracts';

const logger = pino({ name: 'event-bridge' }, process.stderr);

export interface EventMapping {
  /** Event source — uses MqttClient event names, NOT raw MQTT topics.
   *  MqttClient already multiplexes MQTT topics into named events
   *  ('sensor', 'triggerPipeline', 'modeCommand'). The EventBridge
   *  subscribes to these MqttClient events, not to raw MQTT topics.
   *  This avoids the topic-stripping problem: MqttClient.handleMessage()
   *  maps MQTT topics to named events but discards the topic name
   *  before emitting. Filtering by peripheralType/source+stage works
   *  because those fields are preserved in the parsed data. */
  source: 'mqtt:sensor' | 'mqtt:triggerPipeline' | 'mode' | 'custom';
  /** Filter function. Return false to skip this event.
   *  For mqtt:sensor source, filter by peripheralType.
   *  For mqtt:triggerPipeline source, filter by source+stage. */
  filter?: (data: unknown) => boolean;
  /** Named event name forwarded to SpaceManager.handleEvent() */
  eventName: string;
  /** Transform raw data into triggerData for the skill system */
  transform?: (data: unknown) => Record<string, unknown>;
}

export const DEFAULT_EVENT_MAPPINGS: EventMapping[] = [
  // PIR motion → motion_detected
  {
    source: 'mqtt:sensor',
    filter: (data: unknown) => {
      const d = data as { peripheralType?: number; payload?: { motion?: boolean } };
      return d.peripheralType === PERIPHERAL_IDS.PIR && d.payload?.motion === true;
    },
    eventName: 'motion_detected',
    transform: (data: unknown) => {
      const d = data as { payload?: { motion?: boolean } };
      return { motion: true, timestamp: Date.now() };
    },
  },
  // BME280 sensor update → sensor_update
  {
    source: 'mqtt:sensor',
    filter: (data: unknown) => {
      const d = data as { peripheralType?: number };
      return d.peripheralType === PERIPHERAL_IDS.BME280;
    },
    eventName: 'sensor_update',
    transform: (data: unknown) => {
      const d = data as { payload?: Record<string, unknown> };
      return { ...d.payload, timestamp: Date.now() };
    },
  },
  // Voice start → voice_start
  {
    source: 'mqtt:triggerPipeline',
    filter: (data: unknown) => {
      const d = data as { source?: string; stage?: string };
      return d.source === 'voice' && d.stage === 'start';
    },
    eventName: 'voice_start',
    transform: () => ({ timestamp: Date.now() }),
  },
  // Voice end → voice_end
  {
    source: 'mqtt:triggerPipeline',
    filter: (data: unknown) => {
      const d = data as { source?: string; stage?: string };
      return d.source === 'voice' && d.stage === 'end';
    },
    eventName: 'voice_end',
    transform: () => ({ timestamp: Date.now() }),
  },
  // ModeManager transitions → mode_transition
  {
    source: 'mode',
    eventName: 'mode_transition',
    transform: (data: unknown) => {
      const { from, to } = data as { from: string; to: string };
      return { from, to, timestamp: Date.now() };
    },
  },
];

export class EventBridge extends EventEmitter {
  private mappings: Map<string, EventMapping> = new Map();

  constructor(
    private mqtt: MqttClient,
    private spaceManager: SpaceManager,
    private modeManager: ModeManager,
  ) {
    super();
  }

  start(mappings: EventMapping[] = DEFAULT_EVENT_MAPPINGS): void {
    for (const mapping of mappings) {
      this.register(mapping);
    }

    // Wire MqttClient events — subscribe to named events, NOT raw MQTT topics
    this.mqtt.on('sensor', (data: unknown) => {
      this.handleMqttEvent('mqtt:sensor', data);
    });
    this.mqtt.on('triggerPipeline', (data: unknown) => {
      this.handleMqttEvent('mqtt:triggerPipeline', data);
    });

    // Wire ModeManager transitions
    this.modeManager.on('modeChange', ({ from, to }: { from: string; to: string }) => {
      this.forwardModeEvent(from, to);
    });

    logger.info({ count: this.mappings.size }, 'EventBridge started');
  }

  register(mapping: EventMapping): string {
    const id = `${mapping.source}:${mapping.eventName}`;
    this.mappings.set(id, mapping);
    return id;
  }

  /** Route a MqttClient named event to matching EventMappings */
  handleMqttEvent(mqttEventName: 'mqtt:sensor' | 'mqtt:triggerPipeline', data: unknown): void {
    for (const [id, mapping] of this.mappings) {
      if (mapping.source !== mqttEventName) continue;
      if (mapping.filter && !mapping.filter(data)) continue;
      const triggerData = mapping.transform ? mapping.transform(data) : {};
      this.spaceManager.handleEvent(mapping.eventName, triggerData);
      this.emit('event_forwarded', { mappingId: id, eventName: mapping.eventName });
    }
  }

  /** Called by ModeManager when hardware mode transitions */
  private forwardModeEvent(from: string, to: string): void {
    for (const [, mapping] of this.mappings) {
      if (mapping.source !== 'mode') continue;
      const triggerData = mapping.transform ? mapping.transform({ from, to }) : { from, to };
      this.spaceManager.handleEvent(mapping.eventName, triggerData);
      this.emit('event_forwarded', { mappingId: `mode:${mapping.eventName}`, eventName: mapping.eventName });
    }
  }

  /** Add a custom event mapping at runtime (via MCP tool) */
  addCustomMapping(source: EventMapping['source'], eventName: string, filter?: (data: unknown) => boolean, transform?: (data: unknown) => Record<string, unknown>): string {
    const mapping: EventMapping = {
      source,
      eventName,
      filter,
      transform,
    };
    return this.register(mapping);
  }

  removeMapping(id: string): boolean {
    return this.mappings.delete(id);
  }

  listMappings(): EventMapping[] {
    return Array.from(this.mappings.values());
  }

  stop(): void {
    this.mappings.clear();
    logger.info('EventBridge stopped');
  }
}
```

> **Note: Dual MQTT event path with `mcp/events.ts`**
>
> After this change, both `mcp/events.ts` and `EventBridge` handle the same MQTT events:
> 1. **`mcp/events.ts`** → sensor cache updates + MCP notifications to Brain (via `server.notification()`)
> 2. **`EventBridge** → skill event dispatch (via `spaceManager.handleEvent()`)
>
> Both are explicitly called from `core.ts`. This is intentional — they serve different purposes:
> - `mcp/events.ts` is the **Brain notification layer** (pushes events to the connected AI)
> - `EventBridge` is the **skill trigger layer** (activates skills based on events)
>
> Do NOT merge these two systems. They have different lifecycles, different subscribers, and different failure modes.

This is a v1 tool — the Brain can:
- Register a new event mapping (source + eventName, with optional filter/transform)

- List current mappings

The MCP tool handler gets a reference to the `EventBridge` instance (passed via deps).
