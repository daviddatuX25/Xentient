# Plan 07-03: Pack Skill Loader

## Goal
Implement Pack-based skill loading. Packs define CoreSkills in `packs/<name>/skills.json` manifests. The PackLoader validates via Zod, expands to full CoreSkill objects (filling defaults), registers them with `source: 'pack'`, and supports hot-reload on pack switch.

```typescript
import { z } from 'zod';

const PackMetaSchema = z.object({
  name: z.string().min(1),
  version: z.string().min(1),
  description: z.string().optional(),
  author: z.string().optional(),
});

const PackSkillSchema = z.object({
  id: z.string().min(1),
  displayName: z.string().min(1),
  trigger: z.record(z.unknown()), // validated at expansion time
  actions: z.array(z.record(z.unknown())),
  modeFilter: z.string().optional(),
  priority: z.number().int().min(0).max(100).optional(),
  cooldownMs: z.number().int().min(0).optional(),
  escalation: z.record(z.unknown()).optional(),
  collect: z.array(z.record(z.unknown())).optional(),
});

export const PackSkillManifestSchema = z.object({
  pack: PackMetaSchema,
  skills: z.array(PackSkillSchema).min(1),
});
```

Create `harness/src/engine/PackLoader.ts`:

```typescript
import fs from 'fs';
import path from 'path';
import pino from 'pino';
import { PackSkillManifestSchema } from '../shared/contracts-schemas';
import type { CoreSkill, PackSkillManifest, PackSkill, SkillTrigger, CoreAction } from '../shared/types';

const logger = pino({ name: 'pack-loader' }, process.stderr);

export class PackLoader {
  private loadedPack: string | null = null;
  private loadedSkillIds: string[] = [];

  constructor(
    private packsDir: string,
    private registerFn: (skill: CoreSkill) => void,
    private removeFn: (id: string) => boolean,
  ) {}

  /** Load a pack by name. Unloads any previously loaded pack first. */
  loadPack(packName: string): void {
    const manifestPath = path.join(this.packsDir, packName, 'skills.json');

    if (!fs.existsSync(manifestPath)) {
      throw new Error(`Pack manifest not found: ${manifestPath}`);
    }

    const raw = fs.readFileSync(manifestPath, 'utf-8');
    const parsed = JSON.parse(raw);
    const manifest = PackSkillManifestSchema.parse(parsed);

    // Unload previous pack skills (preserve brain/builtin skills)
    this.unloadCurrentPack();

    const expanded: CoreSkill[] = manifest.skills.map(ps => this.expandPackSkill(ps, manifest.pack.name));

    for (const skill of expanded) {
      this.registerFn(skill);
      this.loadedSkillIds.push(skill.id);
    }

    this.loadedPack = packName;
    logger.info({ pack: packName, skillCount: expanded.length }, 'Pack loaded');
  }

  /** Unload the current pack's skills. Brain/builtin skills are preserved. */
  unloadCurrentPack(): void {
    for (const id of this.loadedSkillIds) {
      this.removeFn(id);
    }
    this.loadedSkillIds = [];
    this.loadedPack = null;
  }

  /** Get currently loaded pack name */
  getLoadedPack(): string | null {
    return this.loadedPack;
  }

  /** Expand a PackSkill (simplified) into a full CoreSkill */
  private expandPackSkill(ps: PackSkill, packName: string): CoreSkill {
    return {
      id: ps.id,
      displayName: ps.displayName,
      enabled: true,
      spaceId: '*',  // packs register as global; modeFilter gates behavior
      trigger: ps.trigger as SkillTrigger,
      priority: ps.priority ?? 50,
      actions: ps.actions as CoreAction[],
      collect: ps.collect as CoreSkill['collect'],
      escalation: ps.escalation as CoreSkill['escalation'],
      source: 'pack',
      cooldownMs: ps.cooldownMs ?? 0,
      fireCount: 0,
      escalationCount: 0,
      modeFilter: ps.modeFilter,
      // Tag with pack name for tracking
      ...(packName ? { _pack: packName } : {}),
    };
  }

  /** List available packs in the packs directory */
  listAvailablePacks(): string[] {
    if (!fs.existsSync(this.packsDir)) return [];
    return fs.readdirSync(this.packsDir)
      .filter(dir => fs.existsSync(path.join(this.packsDir, dir, 'skills.json')));
  }

  /** Reload current pack (useful after editing skills.json) */
  reload(): void {
    if (this.loadedPack) {
      const packName = this.loadedPack;
      this.loadPack(packName);
    }
  }
}
```
