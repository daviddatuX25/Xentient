### Step 1: Create SensorHistory ring buffer

**File:** `harness/src/engine/SensorHistory.ts` (NEW, ~60 lines)

```typescript
export interface SensorReading {
  temperature: number | null;
  humidity: number | null;
  pressure: number | null;
  timestamp: number; // Unix epoch ms
}

export class SensorHistory {
  private buffer: SensorReading[] = [];
  private maxEntries: number;
  private lastPushTime = 0;
  private intervalMs: number;

  constructor(maxEntries = 300, intervalMs = 1000) {
    this.maxEntries = maxEntries;
    this.intervalMs = intervalMs;
  }

  push(reading: SensorReading): void {
    const now = Date.now();
    if (now - this.lastPushTime < this.intervalMs) return; // Throttle: 1 entry/second
    this.lastPushTime = now;
    this.buffer.push({ ...reading, timestamp: now });
    if (this.buffer.length > this.maxEntries) this.buffer.shift();
  }

  query(sinceMs?: number): SensorReading[] {
    if (!sinceMs) return [...this.buffer];
    const cutoff = Date.now() - sinceMs;
    return this.buffer.filter(r => r.timestamp >= cutoff);
  }

  clear(): void {
    this.buffer = [];
  }
}
```

### Step 2: Create MotionHistory ring buffer (Expansion 5.4)

The motion timeline needs a history of PIR events. `sensorCache.motion` is only a boolean (last state). Create a ring buffer:

**File:** `harness/src/engine/MotionHistory.ts` (NEW, ~40 lines)

```typescript
export interface MotionEvent {
  timestamp: number;
  active: boolean; // true = motion detected, false = motion cleared
}

export class MotionHistory {
  private buffer: MotionEvent[] = [];
  private maxEntries = 180; // 30 minutes at 10s intervals

  push(active: boolean): void {
    this.buffer.push({ timestamp: Date.now(), active });
    if (this.buffer.length > this.maxEntries) this.buffer.shift();
  }

  query(sinceMs?: number): MotionEvent[] {
    if (!sinceMs) return [...this.buffer];
    const cutoff = Date.now() - sinceMs;
    return this.buffer.filter(e => e.timestamp >= cutoff);
  }
}
```

### Step 3: Create ModeHistory ring buffer (Expansion 5.5)

Mode timeline needs a list of `{mode, startTime, endTime}` intervals. Record mode transitions:

**File:** `harness/src/engine/ModeHistory.ts` (NEW, ~50 lines)

```typescript
export interface ModeInterval {
  mode: string;
  startTime: number;
  endTime: number | null; // null = current mode (still active)
}

export class ModeHistory {
  private intervals: ModeInterval[] = [];
  private maxIntervals = 100;
  private currentMode: string | null = null;
  private currentStart: number = 0;

  recordTransition(mode: string): void {
    const now = Date.now();
    // Close previous interval
    if (this.currentMode) {
      this.intervals.push({ mode: this.currentMode, startTime: this.currentStart, endTime: now });
    }
    // Start new interval
    this.currentMode = mode;
    this.currentStart = now;
    // Trim old entries
    if (this.intervals.length > this.maxIntervals) this.intervals.shift();
  }

  query(sinceMs?: number): ModeInterval[] {
    const result = sinceMs
      ? this.intervals.filter(i => i.endTime === null || i.endTime >= Date.now() - sinceMs)
      : [...this.intervals];
    // Add current mode as open interval
    if (this.currentMode) {
      result.push({ mode: this.currentMode, startTime: this.currentStart, endTime: null });
    }
    return result;
  }
}
```

### Step 5: Sparkline rendering — Canvas with retina DPI (Expansion 5.2)

**File:** `harness/public/js/telemetry.js` (NEW ES module)

```javascript
export class Sparkline {
  constructor(canvasId, maxPoints = 200) {
    this.canvas = document.getElementById(canvasId);
    this.ctx = this.canvas.getContext('2d');
    this.data = [];
    this.maxPoints = maxPoints;
    this.setupDPI();
  }

  setupDPI() {
    const dpr = window.devicePixelRatio || 1;
    const rect = this.canvas.getBoundingClientRect();
    this.canvas.width = rect.width * dpr;
    this.canvas.height = rect.height * dpr;
    this.ctx.scale(dpr, dpr);
    this.canvas.style.width = rect.width + 'px';
    this.canvas.style.height = rect.height + 'px';
  }

  seed(values) {
    this.data = values.slice(-this.maxPoints);
    this.draw();
  }

  push(value) {
    this.data.push(value);
    if (this.data.length > this.maxPoints) this.data.shift();
    this.draw();
  }

  draw() {
    const { ctx, data } = this;
    const w = this.canvas.clientWidth;
    const h = this.canvas.clientHeight;
    ctx.clearRect(0, 0, w, h);

    if (data.length < 2) return;

    const min = Math.min(...data);
    const max = Math.max(...data);
    const range = max - min;

    // Flat-line centering (Expansion 5.3)
    // If all values are identical, center the line vertically
    const step = w / (this.maxPoints - 1);

    ctx.beginPath();
    ctx.strokeStyle = getComputedStyle(this.canvas).getPropertyValue('--sparkline-color') || 'hsl(160, 60%, 45%)';
    ctx.lineWidth = 1.5;

    for (let i = 0; i < data.length; i++) {
      const x = i * step;
      let y;
      if (range === 0) {
        y = h / 2; // Center flat line vertically
      } else {
        y = h - ((data[i] - min) / range) * (h - 4) - 2;
      }
      if (i === 0) ctx.moveTo(x, y);
      else ctx.lineTo(x, y);
    }
    ctx.stroke();
  }
}
```

### Step 10: Mode Timeline rendering

```javascript
function renderModeTimeline(intervals) {
  const container = document.getElementById('mode-timeline');
  const now = Date.now();
  const windowMs = 30 * 60 * 1000; // 30 minutes
  const start = now - windowMs;

  let html = '<div class="mode-timeline-bar">';
  for (const interval of intervals) {
    const left = Math.max(0, ((interval.startTime - start) / windowMs) * 100);
    const right = Math.min(100, (((interval.endTime ?? now) - start) / windowMs) * 100);
    const width = right - left;
    html += `<div class="mode-block mode-${interval.mode}" style="left:${left}%;width:${width}%"></div>`;
  }
  html += '</div>';
  container.innerHTML = html;
}
```

### Step 11: DOM Performance — Batched Updates

```javascript
let pendingUpdates = [];
let updateScheduled = false;

function scheduleUpdate(fn) {
  pendingUpdates.push(fn);
  if (!updateScheduled) {
    updateScheduled = true;
    requestAnimationFrame(() => {
      const fragment = document.createDocumentFragment();
      pendingUpdates.forEach(u => u(fragment));
      pendingUpdates = [];
      updateScheduled = false;
      document.getElementById('skill-log').prepend(fragment);
    });
  }
}
```

## Edge Cases

- **MotionHistory empty on first load:** No dots shown, timeline bar is empty. Acceptable.
- **ModeHistory empty on first load:** No blocks shown. If `modeManager.getMode()` returns a mode, seed one interval from boot time.
- **Canvas DPI resize:** If browser window resizes, canvas needs DPI recalculation. Add `window.addEventListener('resize', ...)` to re-setup DPI. Debounce at 200ms.
- **Sparkline with null values:** Sensor readings can be `null` (sensor not yet reported). Filter nulls before passing to sparkline. If all nulls, show "No data" label.
