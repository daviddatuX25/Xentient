### Step 1: Create SensorHistory ring buffer

**File:** `harness/src/engine/SensorHistory.ts` (NEW, ~60 lines)

```typescript
export interface SensorReading {
  temperature: number | null;
  humidity: number | null;
  pressure: number | null;
  timestamp: number; // Unix epoch ms
}

export class SensorHistory {
  private buffer: SensorReading[] = [];
  private maxEntries: number;
  private intervalMs: number;

  constructor(maxEntries = 300, intervalMs = 1000) {
    this.maxEntries = maxEntries;
    this.intervalMs = intervalMs;
  }

  push(reading: SensorReading): void {
    this.buffer.push({ ...reading, timestamp: Date.now() });
    if (this.buffer.length > this.maxEntries) this.buffer.shift();
  }

  query(sinceMs?: number): SensorReading[] {
    if (!sinceMs) return [...this.buffer];
    const cutoff = Date.now() - sinceMs;
    return this.buffer.filter(r => r.timestamp >= cutoff);
  }

  clear(): void {
    this.buffer = [];
  }
}
```

// GET /api/sensors/history?minutes=5
private async handleGetSensorHistory(req: IncomingMessage, res: ServerResponse): Promise<void> {
  const url = new URL(req.url!, 'http://localhost');
  const minutes = Number(url.searchParams.get('minutes') ?? '5');
  const sinceMs = minutes * 60 * 1000;
  const history = this.deps.sensorHistory.query(sinceMs);
  this.sendJSON(res, 200, history);
}

### Step 4: Sparkline rendering — Canvas

**File:** `harness/public/dashboard.js` (ADD to existing)

```javascript
class Sparkline {
  constructor(canvasId, maxPoints = 200) {
    this.canvas = document.getElementById(canvasId);
    this.ctx = this.canvas.getContext('2d');
    this.data = [];
    this.maxPoints = maxPoints;
  }

  push(value) {
    this.data.push(value);
    if (this.data.length > this.maxPoints) this.data.shift();
    this.draw();
  }

  draw() {
    const { canvas, ctx, data } = this;
    const w = canvas.width;
    const h = canvas.height;
    ctx.clearRect(0, 0, w, h);

    if (data.length < 2) return;

    const min = Math.min(...data);
    const max = Math.max(...data);
    const range = max - min || 1;
    const step = w / (this.maxPoints - 1);

    ctx.beginPath();
    ctx.strokeStyle = getComputedStyle(canvas).getPropertyValue('--sparkline-color') || '#3fb950';
    ctx.lineWidth = 1.5;

    for (let i = 0; i < data.length; i++) {
      const x = i * step;
      const y = h - ((data[i] - min) / range) * (h - 4) - 2;
      if (i === 0) ctx.moveTo(x, y);
      else ctx.lineTo(x, y);
    }
    ctx.stroke();
  }
}
```

### Step 9: DOM Performance — Batched Updates

```javascript
let pendingUpdates = [];
let updateScheduled = false;

function scheduleUpdate(fn) {
  pendingUpdates.push(fn);
  if (!updateScheduled) {
    updateScheduled = true;
    requestAnimationFrame(() => {
      const fragment = document.createDocumentFragment();
      pendingUpdates.forEach(u => u(fragment));
      pendingUpdates = [];
      updateScheduled = false;
      document.getElementById('skill-log').prepend(fragment);
    });
  }
}
```
